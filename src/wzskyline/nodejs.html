          // 【 变量的定义 】

          */
          typeof (typeof 1 == "number") =="boolean";

          var obj ={ name:"test" }; typeof obj =="object"

          /* 【 数据结构 】
          * 常见数据结构 数组 json 对象（也就是map）set
          * 链表和图 可以使用自定义对象来实现
          * 常见方法 数组的十八般武艺, es6 以后并添加了更多的数组操作方法
          */
          var arr =[]
          arr.push(1);
          arr.push({id:1});
          arr.pop()
          arr.join(",")
          JSON.parse(JSON.stringify(arr)) // 可以简单实现数组对象深拷贝
          /* 【 函数 】
          *  fun = function fun (arg){ return "fun";  }
          *  第2个 fun 不写时 第一个fun 就是对其命名, 也就是第一个fun 变量接受了无名函数那 fun 就声明为一个函数 
          *  fun = function fun (arg){ return function () { return "fun2";};  }
          *  fun()() == "fun2"
          *  函数的调用就是这样, js中,函数参数和返回值并不需要声明其类型
          */
 
          /* 【 类 】
          *  js 中的类可以说是一个函数也可以说是一个对象
          *  js 中的继承 一类是原型链上的操作 一类是 extends 关键字的方式 
          */
          class A{}; a = new A()
          typeof A == "function" &&  typeof a == "object" 
          /* 【 运行方式  】
          *  js 是弱语言,非独立的解释性语言
          *  js 是单线程运行的
          *  总体是一行一行代码每个任务去执行的
          *  但是有时候有的代码中间的结果稍后出来 是因为 js运行时 主进程运行简单任务后再到复杂任务队列中取任务进行执行
          *  怎么等到中间代码执行并得到值
          *  async/await Promise
          */
          console.log(1)
          setTimeout(function(){ console.log(2);  },0)
          console.log(3)
          // 这里会看到输出顺序是 1 3 2
          console.log(1)
          async function test() {
            return new Promise(resolve => {
                setTimeout(() => resolve(2), 0);
            });
          }
          console.log(await test()); 
          console.log(3)
          // 这里会看到输出顺序是 1 2 3
