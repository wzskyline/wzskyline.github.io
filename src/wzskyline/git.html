git 常见命令

          
          git reset --hard HEAD && git pull
          git add .
          git commit -m "commit"
          git push
          
          git config --global user.name "wzskyline"
          git config --global user.email "wzskyline@qq.com"
          # 统一去掉账户信息
          git config --system --unset credential.helper
          # 查看用户信息
          git config user.name
          git config user.email
          # 设置信息
          git config --global user.name "your name"
          # 取消全局设置
          git config --global --unset user.name 
          git config --global --unset user.email 
          # 查看日志与项目路径
          git log
          git remote -v 
          # 分支管理
          git branch 
          git branch local
          git checkout local
          git checkout -b local
          #删除分支 
          git branch -d local
          git push --delete origin local
          # 合併至dev
          git add .
          git commit -m "提交"
          git checkout dev     #dev 上需要 是最新的 
          git pull
          git checkout bobi    #bobi分支准备连接到dev
          git rebase dev
          git checkout dev     #dev 上 merge bobi分支
          git merge --no-ff  bobi
          git push origin dev
          git checkout bobi   #bobi分支上与dev分支保持一致 
          git rebase dev
