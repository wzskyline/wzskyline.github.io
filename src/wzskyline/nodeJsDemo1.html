        var transporter = nodemailer.createTransport(config);
        
        // 发送函数
        function send(mail){
            transporter.sendMail(mail, function(error, info){
                if(error) {
                    return console.log(error);
                }
                console.log('mail sent:', info.response);
            });
        };
        
        // 创建一个邮件对象
