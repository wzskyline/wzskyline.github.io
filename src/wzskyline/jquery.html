jquery 中其他内容

        $.ajax({
          //请求方式
          type : "POST",
          //请求地址
          url : "http://127.0.0.1/admin/login",
          //请求的媒体类型
          /* 
          * application/x-www-form-urlencoded 原生 form 表单
          * multipart/form-data               文件数据
          * application/json                  常见json格式
          * text/xml                          文本类型
          */
          headers: {
            "testheader": "test"
          },
          contentType: "application/json;charset=UTF-8",
          //数据，json字符串
          data : JSON.stringify({name:"test",password:"123456"}),
          //请求同步
          async: false,
          //自定义头部
          beforeSend: function (XMLHttpRequest) {  
            XMLHttpRequest.setRequestHeader("SessionID", 1);
          },
          //请求成功
          success : function(result) {
              console.log(result);
          },
          //请求失败，包含具体的错误信息
          error : function(e){
              console.log(e.status);
              console.log(e.responseText);
          },
          //获取响应里面其他信息
          complete: function (xhr, data) { 
            console.log(xhr.getResponseHeader('Date'));  
            console.log(xhr.getAllResponseHeaders()); 
          }
        }); 
        //设置为同步
        $.ajaxSettings.async = false;
        // 简写形式
        $.post("url", data, function(result) {
        
        },"json");
